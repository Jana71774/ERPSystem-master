CREATE TABLE tblLogin (

CREATE PROCEDURE sp_LoginUser
    @LoginId VARCHAR(50),
    @Password VARCHAR(100)
AS
BEGIN
    SET NOCOUNT ON;

    SELECT LoginId, Role
    FROM tblLogin
    WHERE LoginId = @LoginId
      AND PasswordHash = @Password
      AND IsActive = 1
END

11) TransGRN:

11.1) sp_InsertGRN
CREATE   PROCEDURE sp_InsertGRN
    @GRNDate DATE,
    @ItemCode VARCHAR(50),
    @Qty DECIMAL(18,2),
    @BatchCode VARCHAR(50),
    @LoginID VARCHAR(50),
    @VendorName VARCHAR(100),
    @ValueQty DECIMAL(18,2)
AS
BEGIN
    BEGIN TRANSACTION;

    -- 1️⃣ Insert into GRN transaction table
    INSERT INTO transGRN
    (GRNDate, ItemCode, Qty, BatchCode, LoginID, VendorName, ValueQty)
    VALUES
    (@GRNDate, @ItemCode, @Qty, @BatchCode, @LoginID, @VendorName, @ValueQty);

    -- 2️⃣ Update Inventory (Increase AvlQty)
    IF EXISTS (SELECT 1 FROM tblInventory WHERE itemcode = @ItemCode)
    BEGIN
        UPDATE tblInventory
        SET AvlQty = AvlQty + @Qty
        WHERE itemcode = @ItemCode;
    END
    ELSE
    BEGIN
        -- If item is not yet in inventory, create it
        INSERT INTO tblInventory (itemcode, AvlQty, minQty, OrderQty, Description)
        VALUES (@ItemCode, @Qty, 0, 0, '');
    END

    COMMIT TRANSACTION;
END;
